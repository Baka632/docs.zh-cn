## <a name="example"></a><span data-ttu-id="cd743-108">示例</span><span class="sxs-lookup"><span data-stu-id="cd743-108">Example</span></span>

 <span data-ttu-id="cd743-109">下面的示例生成 CS0071。</span><span class="sxs-lookup"><span data-stu-id="cd743-109">The following sample generates CS0071.</span></span>

```csharp
// CS0071.cs
public delegate void MyEvent(object sender);

interface ITest
{
    event MyEvent Clicked;
}

class Test : ITest
{
    event MyEvent ITest.Clicked;  // CS0071

    // Try the following code instead.
    /*
    private MyEvent clicked;

    event MyEvent ITest.Clicked
    {
        add
        {
            clicked += value;
        }
        remove
        {
            clicked -= value;
        }
    }
    */
    public static void Main() { }
}
```
