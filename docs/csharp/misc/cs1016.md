 <span data-ttu-id="f67e8-105">未命名特性参数必须出现在命名参数之前。</span><span class="sxs-lookup"><span data-stu-id="f67e8-105">Unnamed attribute arguments must appear before the named arguments.</span></span>  
  
## <a name="example"></a><span data-ttu-id="f67e8-106">示例</span><span class="sxs-lookup"><span data-stu-id="f67e8-106">Example</span></span>  
 <span data-ttu-id="f67e8-107">以下示例生成 CS1016：</span><span class="sxs-lookup"><span data-stu-id="f67e8-107">The following sample generates CS1016:</span></span>  
  
```csharp  
// CS1016.cs  
using System;  
  
[AttributeUsage(AttributeTargets.Class)]  
public class HelpAttribute : Attribute  
{  
    public HelpAttribute(string url)   // url is a positional parameter  
    {  
        m_url = url;  
    }  
  
    public string Topic = null;      // Topic is a named parameter  
    private string m_url = null;  
}  
  
[HelpAttribute(Topic="Samples", "http://intranet/inhouse")]   // CS1016  
// try the following line instead  
//[HelpAttribute("http://intranet/inhouse", Topic="Samples")]  
public class MainClass  
{  
    public static void Main ()  
    {  
    }  
}  
```
