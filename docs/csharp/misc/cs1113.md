## <a name="example"></a><span data-ttu-id="e372d-110">示例</span><span class="sxs-lookup"><span data-stu-id="e372d-110">Example</span></span>

public static class Extensions  
{  
    public static S ExtMethod(this S s)

        return s;  
    }

{  
}
