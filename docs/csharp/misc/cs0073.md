```csharp  
// CS0073.cs

{  
   public event del MyEvent  
   {  
      add;   // CS0073

      //    MyEvent += value;  
      // }

         MyEvent -= value;  
      }  
  
   }  
  
   public static void Main()  
   {  
   }  
}  
```
