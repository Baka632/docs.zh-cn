```csharp  
{  
   /// text for F(void)  
   void F();

   void F(int i);  
}

{  
   /// <see cref="I.F"/>  
   public static void MyMethod(int i)  
   {  
   }

   /// <see cref="I.F(int)"/>  
   public static void MyMethod(int i)  
   {  
   }  
*/  
   /// text for Main  
   public static void Main ()  
   {  
   }  
}  
```
