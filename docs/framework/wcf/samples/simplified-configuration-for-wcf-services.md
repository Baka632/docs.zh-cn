<configuration>  
  <system.serviceModel>  
    <behaviors>  
      <serviceBehaviors>  
        <behavior name="CalculatorServiceBehavior">

        </behavior>  
      </serviceBehaviors>  
    </behaviors>

      <service name="Microsoft.Samples.GettingStarted.CalculatorService"

      </service>  
    </services>  
  </system.serviceModel>

<configuration>  
  <system.serviceModel>  
    <behaviors>  
      <serviceBehaviors>  
        <behavior name="">

    </behaviors>
